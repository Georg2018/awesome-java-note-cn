# 網頁爬蟲

## 定義
一個程序用於在互聯網中獲取符合指定規則的數據。

## 演示
```
//index.html
<!DOCTYPE html>
<html lang="en">
<head>
	<meta charset="UTF-8">
	<title>Document</title>
</head>
<body>
	abc1@outlook.com
	abc2@outlook.com
	abc3@outlook.com
	abc4@outlook.com
	abc5@outlook.com
</body>
</html>

//RegexTest2.java
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.net.URL;
import java.util.ArrayList;
import java.util.List;
import java.util.regex.Matcher;
import java.util.regex.Pattern;

public class RegexTest2 {

	public static void main(String[] args) throws IOException {

		List<String> list = getMail();

		for (String mail : list) {

			System.out.println(mail);
		}

		List<String> list2 = getMailsByWeb();
		
		System.out.println(list2.size());//獲取符合爬蟲條件的字符串數
	}
	
	//爬取百度首頁上的數據
	public static List<String> getMailsByWeb() throws IOException {

		// 1. 讀取源文件
//		BufferedReader bufr = new BufferedReader(new FileReader("C:\\Users\\asus\\Documents\\index.html"));
		URL url = new URL("https://www.baidu.com");
		
		BufferedReader bufr = new BufferedReader(new InputStreamReader(url.openStream()));

		// 2. 對讀取的數據進行規則的匹配，從中獲取符合規則的數據
		String regex = "\\bbaidu\\b";

		ArrayList<String> list = new ArrayList<>();

		Pattern p = Pattern.compile(regex);

		String line = null;

		while ((line = bufr.readLine()) != null) {

			Matcher m = p.matcher(line);
			while (m.find()) {
				// 3. 將符合規則的數據存儲到集合中
				list.add(m.group());
			}
		}
		return list;
	}

	public static List<String> getMail() throws IOException {

		// 1. 讀取源文件
		BufferedReader bufr = new BufferedReader(new FileReader("C:\\Users\\asus\\Documents\\index.html"));

		// 2. 對讀取的數據進行規則的匹配，從中獲取符合規則的數據
		String mail_regex = "\\w+@\\w+(\\.\\w+)+";

		ArrayList<String> list = new ArrayList<>();

		Pattern p = Pattern.compile(mail_regex);

		String line = null;

		while ((line = bufr.readLine()) != null) {

			Matcher m = p.matcher(line);
			while (m.find()) {
				// 3. 將符合規則的數據存儲到集合中
				list.add(m.group());
			}
		}
		return list;
	}
}
```
